def string_to_days(string):
    my_string = string
    if 'year' not in my_string:
        my_string = '0 years ' + my_string
    if "months" not in string:
        my_string += ' 0 months'
    tokens = my_string.split(' ')
    years = int(tokens[0])
    months = int(tokens[2])
    return years*365 + months*30

import numpy as np
def random_sample(town_data):
    chosen_idx = np.random.choice(len(town_data), replace=True, size=60)
    df_trimmed = town_data.iloc[chosen_idx]
    return df_trimmed

amk = df[df['town']=='ANG MO KIO']
sample = random_sample(amk)
sample.mean() # sample mean resale_pr != population pr
